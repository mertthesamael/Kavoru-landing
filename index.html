Just return

Thin controllers, typed schemas

Routes stay minimal. Validation lives in schemas. Business logic lives in services — the Elysia way. 

import Elysia from "elysia"
import { userSchema } from "../../models/schemas/user"
import { UserService } from "./service"

export default new Elysia({
  name: "users",
  prefix: "/users",
  tags: ["Users"],
})
  .get("/:id", ({ params: { id } }) => UserService.findById(id))
import Elysia from "elysia"
import { authMiddleware } from "../../middleware/auth"

export default new Elysia({ prefix: "/protected" })
  .use(authMiddleware)
  .get("/", ({ user }) => user)
export default new Elysia({ prefix: "/realtime" })
  .ws("/ws", {
    body: wsMessageSchema,
    open(ws) { RealtimeService.register(ws) },
    message(ws, { message }) {
      RealtimeService.handleMessage(ws, message)
    },
    close(ws) { RealtimeService.unregister(ws) },
  })
import { sendEmail } from "../../infra/resend"

await sendEmail({
  to: user.email,
  subject: "Welcome to Kavoru",
  html: "<p>You're all set.</p>",
})

    Quick start

    Up and running in seconds

    Run bunx kavoru@latest, pick features in the terminal (or use flags), then scaffold modules and dev. Every workflow command is one tab away.

    terminal 
    bunx kavoru@latest my-api
# ↑ checkbox picker in terminal (↑↓ Space Enter)
# or: bunx kavoru@latest my-api --features auth,postgres,otel
# or: bunx kavoru@latest my-api --minimal

cd my-api
cp .env.example .env
bun run dev

# → http://localhost:3131
# → http://localhost:3131/help
    bun run dev          # hot reload
bun run start        # prisma generate + run

# compile standalone binary
bun run build        # Windows
bun run build:docker # Linux x64 (Docker)
    bun run kavoru module billing
bun run kavoru module user-profile --force

# creates src/modules/<name>/
#   routes.ts · service.ts · types.ts
# updates routes.registry.ts automatically

bunx kavoru@latest module billing   # same, from any Kavoru project
    bun test

# run a single test file
bun test __tests__/controller.test.ts
    docker compose up --build   # Postgres + auto db push

bunx prisma generate
bunx --bun prisma db push     # sync schema (host dev)

# optional versioned migrations
bunx --bun prisma migrate dev --name init
    bun run otel:view          # trace UI → :4318
bun run otel:tui           # traces in terminal
bun run sentry:spotlight   # errors → :8969

docker compose up --build
docker compose up -d
